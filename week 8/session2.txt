#include <stdio.h>

int main() {

    int a[5] ={1,2,3,4,5};
    
    for (int i = 0; i < 5; i++){
        for (int j = 0 ; j< a[i]; j++){
            printf("*");
        }
        printf("\n");
    }
    
    
    //Output:
    // *
    // * *
    // * * *
    // * * * *
    // * * * * *
   

    return 0;
}
